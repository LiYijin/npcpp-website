代码示例

简单的数据包处理

package main;

import packet;
import filter;

int main() {
    Packet p = recv();

    // 应用防火墙规则
    if (filter.allow(p.src_ip, p.dst_port)) {
        p.forward("allowed");
    } else {
        p.drop();
    }

    return 0;
}

负载均衡器

package loadbalancer;

import packet;
import pool;

int main() {
    Packet p = recv();
    Server s = pool.select_server();

    if (s.is_healthy()) {
        p.forward(s.address);
        p.mark_processed();
    } else {
        p.forward("backup_pool");
    }

    return 0;
}
